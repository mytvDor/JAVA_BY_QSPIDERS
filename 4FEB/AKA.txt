class LoopEx {
   public static void main (String [] args){

for(int i=0 ;i<=10 ;i++ ){

System.out.println(i);

}

for(char i='a' ;i<='z' ;i++ ){

System.out.println(i);

}


for(char i='A' ;i<='Z' ;i++ ){

System.out.println(i);

}


for(char i='0' ;i<='9' ;i++ ){

System.out.println(i);

}


for(int i=0 ;i<=127 ;i++ ){

System.out.println((char)i);

}
}
}
